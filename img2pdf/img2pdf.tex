\newcommand\includeFullPageGraphics[2][]{
  \newpage
  \thispagestyle{empty}
  \begin{tikzpicture}[remember picture, overlay, inner sep=0pt]
    \node at (current page.center) 
      {\includegraphics[width=\paperwidth, height=\paperheight, keepaspectratio=false,#1]{#2}};
  \end{tikzpicture}
  \newpage
}
